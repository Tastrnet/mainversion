DROP FUNCTION IF EXISTS get_nearby_restaurants(float, float, int, int); CREATE OR REPLACE FUNCTION get_nearby_restaurants( lat float, lng float, radius int DEFAULT 5000, limit_count int DEFAULT 20 ) RETURNS TABLE( id int, name text, address text, latitude numeric, longitude numeric, cuisines jsonb, is_featured boolean, distance_meters float ) LANGUAGE plpgsql STABLE AS $$ BEGIN RETURN QUERY SELECT r.id, r.name, r.address, r.latitude, r.longitude, r.cuisines, r.is_featured, ST_Distance( r.geom::geography, ST_SetSRID(ST_Point(lng, lat), 4326)::geography ) AS distance_meters FROM restaurants r WHERE r.geom IS NOT NULL AND ST_DWithin( r.geom::geography, ST_SetSRID(ST_Point(lng, lat), 4326)::geography, radius ) ORDER BY r.geom <-> ST_SetSRID(ST_Point(lng, lat), 4326) LIMIT limit_count; END; $$;
